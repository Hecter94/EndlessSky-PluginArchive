mission "Hire Interceptor Company"
	name "Hire Interceptor Company"
	description "The Quicksilver Corporation is offering a small fleet of interceptors as escorts for an upfront payment of 50,000 credits and a daily cost of 900 for a period of 3 months."
	repeat
	deadline 90
	job
	to offer
		not "Hire Interceptor Company: active"
		not "Hire Interceptor Company2: active"
		random < 100
		"combat rating" > 200
	source
		government Republic "Free Worlds" Syndicate Neutral
	destination
		government Republic "Free Worlds" Syndicate Neutral
		distance 1 1
	npc
		government "Quicksilver CO"
		personality disables
		fleet
			names "civilian"
			variant
				"Silver Stream" 3
	to complete 
		never
	on fail
		"salary: Mercenaries" += 900
		dialog "The fleet leaves your service as the contract has expired."
	on accept
		payment -50000
		"salary: Mercenaries" -= 900
		dialog "The captain of the fleet shakes your hand as you sign the document and states 'The Quicksilver Coperation always guarantees a safe voyage with our escorts at your side!' before boarding his ship."
