shipyard "Thule Bare"
	"Hekla Dora (Bare)"
	"Hekla Vaba (Bare)"


mission "Factory Outlets: DAIS installed"
	landing
	invisible
	non-blocking
	repeat
	to offer
		has "installed plugin: DAIS"
		not "event: factory outlets: dais installed"
	on offer
		event "factory outlets: dais installed"
		clear "event: factory outlets: dais uninstalled"
		fail

event "factory outlets: dais installed"
	planet Thule
		add shipyard "Thule Bare"
	shipyard "Lionheart Bare"
		add "Lindwurm (Bare)"
	shipyard "Megaparsec Bare"
		add "Hydrargyrum (Bare)"

mission "Factory Outlets: Erinyes Bare"
	landing
	invisible
	to offer
		has "event: erinyes for sale"
	on offer
		event "factory outlets: erinyes bare"
		fail

event "factory outlets: erinyes bare"
	shipyard "Southbound Bare"
		add "Erinys (Bare)"


mission "Factory Outlets: DAIS uninstalled"
	landing
	invisible
	non-blocking
	repeat
	to offer
		not "installed plugin: DAIS Ships"
		has "event: factory outlets: dais installed"
	on offer
		event "factory outlets: dais uninstalled"
		clear "event: factory outlets: dais installed"
		fail

event "factory outlets: dais uninstalled"
	planet Thule
		remove shipyard "Thule Bare"
	shipyard "Lionheart Bare"
		remove "Lindwurm (Bare)"
	shipyard "Megaparsec Bare"
		remove "Hydrargyrum (Bare)"
	shipyard "Southbound Bare"
		remove "Erinys (Bare)"
