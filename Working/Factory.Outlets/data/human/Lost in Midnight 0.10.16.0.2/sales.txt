mission "Factory Outlets: LiM installed"
	landing
	invisible
	non-blocking
	repeat
	to offer
		has "installed plugin: Lost in Midnight"
		not "event: factory outlets: lim installed"
	on offer
		event "factory outlets: lim installed"
		clear "event: factory outlets: lim uninstalled"
		fail

event "factory outlets: lim installed"
