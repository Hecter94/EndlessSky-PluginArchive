

mission `MGF: Mobile Outfitter Deploy (Aya'k'k)`
	repeat
	landing
	name `Deploy mobile outfitter`
	to offer
		has "ship model: oU923 Isoxys"
	source
		planet "MGF: Orbit Point A (Aya'k'k)"
		attributes "orbit point"
	on offer
		conversation
			#would be neat if I can get the name of the mobile outfitter ship
			`Deploy the oU923 Isoxys as an outfitter station here? WARNING: Any custom outfitting on the ship will be GONE FOREVER. Also make sure you have other ship to leave with.`
			choice
				`	(Yes)`
				`	(Not yet)`
					defer
			action
				event "mgf: deploy mobile outfitter (Aya'k'k)" 0
				take ship "oU923 Isoxys"
				#Maybe take-ship should be able to set id for game to memorize it as well for giving the ship back to the player.
			`Done, you can undeploy by landing again and going to the job board. (If you cannot see the outfitter, reload the current save or depart and land again)`
			branch give_shuttle
				"total ships" == 0
			``
				decline
			label give_shuttle
			action #Not gonna bother to take this ship back
				give ship "Shuttle" "Escape Pod 0"
					id "mgf: isoxys emergency shuttle"
			`Since you don't have another ship, you took one of the emergency Shuttle from the ship.`
				decline

mission `MGF: Mobile Outfitter Undeploy (Aya'k'k)`
	repeat
	job
	name `Undeploy mobile outfitter`
	description `Accept this job to undeploy your oU923 Isoxys and return the ship to your fleet. (Take any outfit you stored here into cargo first if you don't want to lose it)`
	source
		planet "MGF: Orbit Point A (Aya'k'k)"
		attributes "mgf: mobile shipyard"
	on offer
		event "mgf: undeploy mobile outfitter (Aya'k'k)" 0
			#would be neat of we can memorize ship name taken from the player as well as the loadout to re-apply.
		give ship "oU923 Isoxys" "Isoxys"
	on accept
		fail
	on enter
		fail

event "mgf: deploy mobile outfitter (Aya'k'k)"
	planet "MGF: Orbit Point A (Aya'k'k)"
		#Would be neat if we can just pull the name of the ship and put it here.
		"display name" "Isoxys"
		government "MGF: Player Company"
		landscape `land/space0`
		attributes "mobile shipyard" "mgf: mobile shipyard"
		description `Your Isoxys-class mobile shipyard is currently deployed here. It's vast array turrets provides decent security allowing you to service your ships here in peace.`
		outfitter "MGF tbab Player Basic" 
		port `Dock`
			recharges all
			services
				`job board`
			description `You can refuel and refit your ship here.`
		remove to know
	system "Aya'k'k"
		remove object "MGF: Orbit Point A (Aya'k'k)"
			sprite "planet/mgf orbit point"
			distance 5000
			period 1200
		add object "MGF: Orbit Point A (Aya'k'k)"
			sprite "planet/mgf hh isoxys station"
				"scale" .5
			distance 5000
			period 1200

event "mgf: undeploy mobile outfitter (Aya'k'k)"
	planet "MGF: Orbit Point A (Aya'k'k)"
		"display name" "Orbit Point"
		remove government "MGF: Player Company"
		add government "MGF: Orbit Point"
		remove landscape `land/space0`
		add landscape `land/mgf actuallyspace0 landscape`
		remove attributes "mobile shipyard" "mgf: mobile shipyard"
		add attributes "orbit point"
		description  `A location in orbit where you can deploy your mobile outfitter or mobile shipyard.`
		remove outfitter
		remove port
		# to know #wait for 0.10.17
		# 	"ship model: oU923 Isoxys" >= 1
	system "Aya'k'k"
		remove object "MGF: Orbit Point A (Aya'k'k)"
			sprite "planet/mgf hh isoxys station"
				"scale" .5
			distance 5000
			period 1200
		add object "MGF: Orbit Point A (Aya'k'k)"
			sprite "planet/mgf orbit point"
			distance 5000
			period 1200

planet "MGF: Orbit Point A (Aya'k'k)"
