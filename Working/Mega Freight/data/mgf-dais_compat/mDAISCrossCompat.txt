mission "MGF-DAIS trigger"
	landing
	invisible
	repeat
	to offer
		# has "mgf dev mode"
		or
			"MGF-DAIS deactivation: offered" >= "MGF-DAIS trigger: offered"
			not "MGF-DAIS trigger: offered"
		or
			has "installed plugin: DAIS"
	on offer
		event "mgf-dais: cross-compat activate"
		"mgf-dais compat active" = 1
	on accept
		fail
