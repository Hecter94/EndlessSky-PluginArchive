mission "Rumskib: DAIS installed"
	landing
	invisible
	non-blocking
	repeat
	to offer
		has "installed plugin: DAIS"
		not "event: rumskib: DAIS installed"
	on offer
		event "rumskib: DAIS installed"
		clear "event: rumskib: DAIS uninstalled"
		fail
