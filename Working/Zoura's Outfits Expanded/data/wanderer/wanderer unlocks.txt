# Unlocks the Effusebeam

mission "Outfits Expanded: Effusebeams Available"
	landing
	invisible
	to offer
		or
			has "event: pond strider mass production v2"
			has "event: pond strider mass production"
	on offer
		event "outfits expanded effusebeams available" 90

event "outfits expanded effusebeams available"
	outfitter "Wanderer Basics"
		"Effusebeam"
	outfitter "Wanderer Advanced"
		"Effusebeam"
