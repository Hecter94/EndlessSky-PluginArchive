mission "ATHI: Player Conquer Hamuhtlat"
	landing
	invisible
	source "Hamuhtlat"
	name "Conquer Hamuhtlat"
	description `Retake Hamuhtlat`
	to offer
		has "event: ATHI: Athmagr conquer 1"
		has "tribute: Hamuhtlat"
	on offer
		conversation
			`You've chased away the Athmagr stationed here but to hold it you'll need a fleet to defend it, check the job board to see what fleet you can buy.`
				accept
		event "ATHI: Player conquer Hamuhtlat"
		clear "ATHI: Athmagr Reconquer Hamuhtlat notice: offered"
	on accept
		fail
		
mission "ATHI: Athmagr Reconquer Hamuhtlat notice"
	landing
	repeat
	source
		not planet "Hamuhtlat"
	to offer
		"ATHI: defense counter: Hamuhtlat" <= 0
		random < 35
		not "ATHI: Athmagr Reconquer Hamuhtlat notice: offered"
		has "ATHI: Player Conquer Hamuhtlat: offered"
	on offer
		conversation
			`NOTICE: Your fleet defending Hamuhtlat have been defeated and the planet was taken over by the Athmagr.`
			`	[note: map will update after departing. Relinquish Tribute and Demand Tribute again to reconquer it.]`
				decline
		event "ATHI: Athmagr reconquer Hamuhtlat"
		clear "ATHI: Player Conquer Hamuhtlat: offered"
		clear "tribute: Hamuhtlat"
				
mission "ATHI: Athmagr Reconquer Hamuhtlat warning"
	landing
	repeat
	source "Hamuhtlat"
	to offer
		"ATHI: defense counter: Hamuhtlat" <= 0
		has "ATHI: Player Conquer Hamuhtlat: offered"
	on offer
		conversation
			`WARN: Your do not have any active fleet defending <origin>, Athmagr will come back and take it if you leave it undefended.`
				decline

mission "ATHI: Hamuhtlat Defense [Militia]"
	job
	repeat
	source "Hamuhtlat"
	name `Hamuhtlat Defense (Militia)`
	description `Buy very large militia fleet consist of ships from Southbound and Tarazed shipyards to defend Hamuhtlat for <payment>.`
	to offer
		#"net worth" >= 100000000
		has "tribute: Hamuhtlat"
	on accept
		payment -75000000
		"ATHI: defense counter: Hamuhtlat" += 1
	# Used kill to require all ships to be dead, save requires only 1 ship dead to fail.
	npc kill
		government "Escort"
		personality heroic uninterested staying
		fleet
			names "player navy"
			fighters "player fighters"
			variant
				"Falcon (Heavy)" 5
				"Falcon" 5
				"Bastion (Heavy)" 7
				"Argosy (Missile)" 12
				"Argosy (Turret)" 12
				"Osprey (Missile)" 7
				"Fury" 12
		conversation #Required because on complete have specific destination and save don't work, so conversation is used instead to trigger the value reduction.
			`Your Hamuhtlat militia defense fleet has been wiped out!`
			action
				"ATHI: defense counter: Hamuhtlat" -= 1
	on abort
		payment 75000000
		"ATHI: defense counter: Hamuhtlat" -= 1
