mission "Music: Kestrel Testing"
    invisible
    landing
    non-blocking
    to offer
        not "Kestrel Testing: done"
        or
            "combat rating" > 6000
            and
                "combat rating" > 2000
                has "global: unlocked kestrel"
    to accept
        has "Kestrel Testing: active"
    to complete
        never
    on accept
        event "kestrel testing music"
    on enter "Umbral"
        dialog `The music you are hearing seems familiar, like it was in some kind of dream of yours. After some thinking, you come to the conclusion that it was not another escape velocity dream.`
        fail
