# Makes missile launchers fire like their appearances would suggest.
# e.g. Torpedo Launcher has two tubes, so it fires two torpedoes at once.
# Weapon DPS remains the same, this just shifts missiles away from a DPS orientation to a burst-damage one.

# "burst reload" defaulting to 15 is based on Finisher Maegrolain setting a rough precedent for what value to use here without using submunitions to literally fire everything on a single trigger pull.

# Not currently done for pod size launchers as they already use the burst stats, so would be more involved, needing submunition definitions.



outfit "Meteor Missile Launcher"
	weapon
		"burst count" 3
		"burst reload" 15

outfit "Sidewinder Missile Launcher"
	weapon
		"burst count" 4
		"burst reload" 20 # slower since they rotate out on a cylinder

outfit "Torpedo Launcher"
	weapon
		"burst count" 2
		"burst reload" 15

outfit "Typhoon Launcher"
	weapon
		"burst count" 3
		"burst reload" 15

outfit "Heavy Rocket Launcher"
	weapon
		"burst count" 3
		"burst reload" 15



outfit "Piercer Missile Launcher"
	weapon
		"burst count" 7
		"burst reload" 10
		"stream" # needs to be re-added if weapon is modified

outfit "Korath Piercer Launcher" # Old name for backwards compatibility.

		"stream" # needs to be re-added if weapon is modified
