outfitter "Star Tail Access"
	to sell
		has "gamerule: universal ammo restocking"
		not "license: Star Tail Tech"
	location
		attributes "outfitter"
		outfits "Star Tail Cell"
	stock
		"Star Tail Access Temporary License"

outfitter "Star Tail Cell Restock"
	to sell
		or
			has "license: Star Tail Access Temporary"
			has "license: Star Tail Tech"

mission "Midnight Tech: Star Tail Access Permanent"
	entering
	invisible
	destination Earth
	to offer
		has "license: Star Tail Access Temporary"
		not "license: Star Tail Tech"
	on offer
		clear "license: Star Tail Access Temporary"
		set "license: Star Tail Tech"
		fail

mission "Midnight Research: Star Tail Tech"
	entering
	invisible
	destination Earth
	to offer
		has "outfit (flagship installed): Star Tail Cell"
		has "outfit (flagship installed): Ammo Basics"
		has "outfit (flagship installed): Research Data"
		has "outfit (flagship installed): Research Data (Bio)"
		has "outfit (flagship installed): Research Data (Chem)"
		has "outfit (flagship installed): Research Data (Phys)"
		has "license: Ammo Expertise"
		not "license: Star Tail Access Temporary"
		not "license: Star Tail Tech"
		random < 1 + ( "flagship attribute: research bonus" / 10 )
	on offer
		outfit "Research Data" -1
		outfit "Research Data (Bio)" -1
		outfit "Research Data (Chem)" -1
		outfit "Research Data (Phys)" -1
		set "license: Star Tail Tech"
		conversation
			scene "outfit/tech star tail cell"
			`You have successfully completed Star Tail Tech research. This makes it possible for any outfitter to offer Star Tail Cells even if it is normally not stocked there."`
			``
			`	(Lost 1 Research Data)`
			`	(Lost 1 Research Data (Bio))`
			`	(Lost 1 Research Data (Chem))`
			`	(Lost 1 Research Data (Phys))`
			`	(Gained Star Tail Tech)`
				decline
