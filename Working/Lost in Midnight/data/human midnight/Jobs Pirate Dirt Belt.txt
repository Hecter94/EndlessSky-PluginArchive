	name "Delivery to <planet>"
	job
	repeat
	description "Deliver <tons> of cargo to <destination>. Payment is <payment>."

	to offer
		random < 10 + "Sketchy Cargo [0]: done" + "Sketchy Cargo [1]: done"

	destination
		distance 2 10

		payment 0 160
