mission "Bounty Hunting: Mother Goose"
	job
	deadline 14
	name "Wanted Bandit near <system>"
	description "An old Aerie named the Mother Goose and its fighters have been attacking merchants near <system>. Destroy them by <day> for payment (<payment>)."
	"apparent payment" 150000
	source
		government "Republic"
		attributes "deep"
	to offer
		"combat rating" >= 250
		random < 10
	npc kill
		government "Bounty"
		system
			distance 1 1
		personality target waiting staying plunders harvests
			confusion 20
		ship "Aerie (Energy Cannons)" "Mother Goose"
		ship "Dagger" "Jack"
		ship "Dagger" "Jill"
		dialog `The Mother Goose and their fleet have been eliminated. Shortly afterwards, the government of <planet> sends you <payment> along with a message of gratitude.`
		on kill
			payment 150000
			fail

mission "Engineering Room: Prototype Test 1"
	name "Prototype Test"
	job
	description "A prototype will be installed on your ship. Disable the robotic target dummy ship in <system> then land on <planet>. Warning: the target ship is programmed to engage in self-defense behaviors to ensure a realistic simulation. Payment is <payment>."
	source
		government "Republic" "Syndicate" "Free Worlds" "Independent" "Neutral" "Pirate"
		attributes "research"
		not attributes "outfitter"
	to offer
		random < 15
		"combat rating" > 1000
	on offer
		require "Engineering Room"
	on accept
		outfit "Prototype Ramscoop FCX" 1
		dialog `The researchers install the Ramscoop FCX onto your ship. There are still wires jutting out from where it's installed...`
	npc save disable
		government "Test Dummy"
		personality disables staying heroic target
			confusion 120
		system destination
		ship "Firebird Test Dummy" "Test Vessel"
		dialog "Somehow, you survived. This prototype needs some serious rework. The sooner you get it off your ship, the better!"
	on visit
		dialog "You forgot to disable the target dummy ship in <system>! You'll need to head back there and do that before you can return here for payment."
	on abort
		outfit "Prototype Ramscoop FCX" -1
		clear "Engineering Room: Prototype Test 2: offered"
	on fail
		outfit "Prototype Ramscoop FCX" -1
		clear "Engineering Room: Prototype Test 2: offered"
		dialog "The robotic target ship has been destroyed. You are hailed by an angry researcher who fines you <payment>."
		payment -200000
	on complete
		payment 200000
		outfit "Prototype Ramscoop FCX" -1
		clear "Engineering Room: Prototype Test 2: offered"
		dialog "The Ramscoop FCX is uninstalled from your ship. You collect your payment of <payment> while your engineer gives the researchers the harsh truth about their prototype."
