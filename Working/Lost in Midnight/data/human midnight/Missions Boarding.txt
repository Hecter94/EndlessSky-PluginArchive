				`	(Refuse to take it.)`
					decline
			`	You accept <cargo> onto the <ship>. Deliver it to <destination> for <payment>.`
				accept
	on complete

mission "Medical Bay: Injured Crewmember (Coalition)"
	assisting
	deadline
	repeat
	minor
	name "Injured Crewmember"
	description "Bring an injured crewmember to <destination>."
	passengers 1
	source
		government "Coalition" "Heliarch"
		not attributes "automaton"
	destination
		government "Coalition" "Heliarch"
		attributes spaceport
		not attributes "uninhabited"
		distance 1 3
	to offer
		has "known to the heliarchs"
		has "outfit (flagship installed): Medical Bay"
		random < 5
	on offer
#		require "Medical Bay"
		conversation
			scene "outfit/medical bay"
			`You've helped repair the <origin>, but one of their crewmembers was critically injured and needs to be transported in your medical bay to a nearby hospital by <date> for adequate treatment.`
				accept
	on fail
		dialog `You failed to deliver the injured crewmember from the <origin> to a hospital in time...`
	on complete
		"reputation: Coalition" ++
		"reputation: Heliarch" ++
		dialog `You inform the spaceport that you have an injured crewmember from the <origin> who needs medical attention. An emergency transport arrives shortly and whisks the injured crewmember away.`
