mission "Midnight Dust Cloud (Astral)"
	landing
	invisible
	deadline 2
	repeat
	source
		attributes "midnight cloud astral" "midnight cloud astral spawn"
	to offer
		has "flagship attribute: atmosphere scan"
		random < 90 - ( "midnight dust scan" * 5 )
	on offer
		clear "midnight dust scan default"
		clear "midnight dust scan low"
		clear "midnight dust scan high"
		clear "midnight dust scan"
		conversation
			`This cloud of cosmic dust may have hidden secrets. Survey it with your atmosphere scanner?`
			label begin
			choice
				`	(Scan using the default frequency.)`
					goto scan-default
					to display
						not "midnight dust scan default"
				`	(Scan using a low frequency.)`
					goto scan-low
					to display
						not "midnight dust scan low"
				`	(Scan using a high frequency.)`
					goto scan-high
					to display
						not "midnight dust scan high"
				`	(Leave it alone.)`
			`	You've decided to avoid exploring the cloud for now...`
				defer
			label scan-default
			action
				set "midnight dust scan default"
				"midnight dust scan" ++
			``
				goto scan
			label scan-low
			action
				set "midnight dust scan low"
				"midnight dust scan" ++
			``
				goto scan
			label scan-high
			action
				set "midnight dust scan high"
				"midnight dust scan" ++
			``
			label scan
			branch bioroid
				has "flagship planet attribute: midnight cloud astral spawn"
				random < 15
			branch natural
				has "flagship planet attribute: midnight cloud astral"
				random < 15
			branch research
				random < 35 + ( "midnight dust scan" * 5 )
			scene "outfit/unknown"
			branch full-frequency
				has "midnight dust scan default"
				has "midnight dust scan low"
				has "midnight dust scan high"
			`	Your atmosphere scans of the dust cloud are inconclusive. You could try scanning with a different frequency or entering the cloud again from a different angle to study it some more.`
				goto begin
			label full-frequency
			`	After using all frequencies, your atmosphere scans of the dust cloud are inconclusive. Perhaps you could try entering the cloud again from a different angle to study it some more.`
				defer
			label research
			branch physics
				random < 5
			branch chemistry
				random < 10
			branch biology
				random < 15
			action
				outfit "Research Data" 1
			scene "outfit/research"
			`	Your atmostphere scans form a collection of data from the cloud of cosmic dust. This may have some scientific value.`
				decline
			label biology
			action
				outfit "Research Data (Bio)" 1
			scene "outfit/research biology"
			`	Your atmostphere scans detect unidentified DNA strands floating within the cloud of cosmic dust. This may have some scientific value.`
				decline
			label chemistry
			action
				outfit "Research Data (Chem)" 1
			scene "outfit/research chemistry"
			`	Your atmostphere scans detect unidentified molecule structures floating within the cloud of cosmic dust. This may have some scientific value.`
				decline
			label physics
			action
				outfit "Research Data (Phys)" 1
			scene "outfit/research physics"
			`	Your atmostphere scans detect unidentified energy waves within the cloud of cosmic dust. This may have some scientific value.`
				decline
			label natural
			branch spores
				random < 5
			branch pollen
				random < 20
			action
				outfit "Deuterium" 1
			scene "outfit/harvested deuterium"
			`	You find a large chunk of frozen deuterium floating in the cosmic dust. This dust cloud may have been a proto-comet.`
				decline
			label pollen
			action
				outfit "Pollen?" 3
			scene "outfit/pollen particle"
			`	You find strange protein particles. The atmosphere scanner identifies them as some sort of pollen. Perhaps they were spread throughout the galaxy by a form of interstellar plantlife.`
				decline
			label spores
			action
				outfit "Murkspores?" 10
			scene "outfit/murkspores"
			`	You find strange protein particles. The atmosphere scanner identifies them as some sort of spores. Perhaps they were spread throughout the galaxy by a form of interstellar fungus.`
				decline
			label bioroid
			scene "scene/scene-bioroid"
			`	A bioroid is gently floating in the dust cloud. Take it into your cargo bay?`
			choice
				`	(Catch it.)`
				`	(Leave it.)`
					decline
			scene "scene/scene-bioroid crushed"
			action
				outfit "Bioroid (Crushed)"
			branch mommy
				random < 10
			`	You position the <ship> to catch the bioroid. Once it is inside, you shut the cargo bay doors. It seems to be reacting to the pressurized air...`
				decline
			label mommy
			`	You position the <ship> to catch the bioroid. Once it is inside, you shut the cargo bay doors. It seems to be reacting to the pressurized air...`
				accept
	on accept
		clear "midnight dust scan default"
		clear "midnight dust scan low"
		clear "midnight dust scan high"
	on decline
		clear "midnight dust scan default"
		clear "midnight dust scan low"
		clear "midnight dust scan high"
	on defer
		clear "midnight dust scan default"
		clear "midnight dust scan low"
		clear "midnight dust scan high"
	on enter
		dialog "This creature doesn't seem happy to see you..."
	npc
		government "Indigenous Lifeform (Hostile)"
		personality target waiting mute heroic vindictive
		fleet "Astral"
