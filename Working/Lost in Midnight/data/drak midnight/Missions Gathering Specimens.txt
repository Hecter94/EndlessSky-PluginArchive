mission "Gathering Specimens: Alexandria"
	minor
	name "Gathering Specimens: Alexandria"
	description "Search the galaxy for interstellar life. Bring tissue samples back to <origin>."
	source Alexandria
	to offer
		or
			"Transport scientists: done" > 5
			has "Deep: Project Hawking: done"
		or
			and
				not "Randomize Gathering Specimens: Alexandria"
				random < 70
			and
				has "Randomize Gathering Specimens: Alexandria"
				random < 25
	to complete
		never
	on offer
		conversation
			branch repeat
				has "Gathering Specimens: Alexandria: failed"

			`A group of scientists approach you. "Captain <first> <last>? We've been told that you're a reliable captain who travels beyond The Deep. If that's true, would be interested in helping us with our research?"`
			choice
				`	"What sort of help do you need?"`
			`	"Nothing too hard. Just keep an eye out for any interstellar life forms. If you find one, bring us back a tissue sample. We'll make sure you are compensated."`
				goto end

			label repeat
			`	The scientists you previously delivered specimens to are requesting more.`

			label end
			choice
				`	(Accept their request.)`
					accept
				`	(Decline their request.)`
					defer

	on defer
		set "Randomize Gathering Specimens: Alexandria"
	on accept
		set "Randomize Gathering Specimens: Alexandria"
