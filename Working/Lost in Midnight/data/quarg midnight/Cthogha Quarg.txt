mission "Midnight Quarg Cthogha First Contact"
	transition
	invisible
	name "Blood Stone"
	description `Harvest a "blood stone" from a "sky stone" and deliver it to <destination>.`
	source
		system "Quarto"
	destination "Quarry"
	on offer
		set "language: Quarg (Cthogha)"
		conversation
			scene "planet/station0old"
			`The Quarg station in this system hails you. They do not speak your language, but after studying you they send a written message. The cultural data archive from Alexandria indicates a 66% match with ancient Cuneiform! Perhaps this is a regional dialect or a predecessor to it. The message roughly translates as: "We know you home. You walk far. We in trap. You road home?"`
			choice
				`	"I had blood stone. Door open. Stone is key."`
			scene "outfit/bloodcore"
			`	"We dig sky stone. No find blood stone. You bring blood stone. We trade." The transmission ends.`
				accept
	to complete
		has "outfit: Quantum Bloodcore"

mission "Midnight Quarg Cthogha Quantum Bloodstone"
	minor
	repeat
	name "Quantum Bloodstone"
	description "Bring two Quantum Bloodcores to <destination>. The Quarg will keep one and modify the remaining one for you."
	source "Quarry"
	to offer
		has "language: Quarg (Cthogha)"
		has "outfit: Quantum Bloodcore"
		random < 70 + "outfit: Quantum Bloodcore"
		"quarry bloodstone" < 5
	on offer
		conversation
			scene "outfit/bloodcore"
			`A some Quarg approach you. "You dig blood core? Bring more here. We make blood stone.`
			choice
				`	"I bring two blood core. You make blood stone."`
					accept
					to display
						"outfit: Quantum Bloodcore" < 2
				`	"I have two blood core. You make blood stone."`
					to display
						"outfit: Quantum Bloodcore" >= 2
				`	"No want blood stone. I keep blood core."`
					decline
			action
				outfit "Quantum Bloodcore" -2
				outfit "Quantum Bloodstone" 1
				"quarry bloodstone" ++
			scene "outfit/bloodstone"
			`The Quarg keep one Quantum Bloodcore and modify the other one into a Quantum Bloodstone.`
				decline
	to complete 
		"outfit: Quantum Bloodcore" >= 2
	on complete
		outfit "Quantum Bloodcore" -2
		outfit "Quantum Bloodstone" 1
		"quarry bloodstone" ++
		conversation
			scene "outfit/bloodstone"
			`You have delivered two Quantum Bloodcores. The Quarg keep one and modify the other one into a Quantum Bloodstone.`

mission "Midnight Quarry Typhon Refugee"
	minor
	clearance
	name "Typhon Refugees"
	description "Escort the <npc> and its fleet to <destination>"
	source Quarry
	destination "Alta Hai"
	to offer
		"quarry bloodstone" >= 3
		random < 70
	on offer
		conversation
			`A group of Quarg approach. "We have much blood stone. Want go home. You lead way?"`
			choice
				`	"Yes, I lead way home."`
				`	"No, not now."`
					defer
			`	The Quarg turn several shades of pastel colors in response and eagerly prepare to depart.`
				accept
	on accept
		"quarry bloodstone" -= 3
	npc save accompany
		government "Quarg (Cthogha)"
		personality target escort timid
		fleet
			names "quarg"
			cargo 3
			commodities "Food" "Equipment" "Electronics"
			variant
				"Quarg Typhon (Bloodstone)"
				"Quarg Naga (Bloodstone)" 2
	on complete
		"reputation: Quarg" += 5
		"reputation: Quarg (Hai)" += 25
		"reputation: Quarg (Kor Efret)" += 5
		"reputation: Quarg (Gegno)" += 5
		"reputation: Quarg (Incipias)" += 5
		"reputation: Quarg (Cthogha)" += 50
		outfit "Quantum Bloodstone" 1
		conversation
			`The Quarg of <planet> were surprised when you arrived but have welcomed the refugees from Quarry. The Quarg from Quarry are happy to finally meet other Quarg. The captain of the <npc> says something to a representative of the ringworld, who flashes a dark shade of purple. It responds to the captain, who turns blue.`
			choice
				`	"Is something wrong?"`
			scene "outfit/bloodstone"
			`	The ringworld Quarg turns to you. "Your friends were so grateful for your assistance that they wanted to give you one of their ships. Obviously, we cannot allow that and have informed them of this fact. However, we have identified a foreign object on one of their ships and hope you will accept it as a token of gratitude." It is a Quantum Bloodstone. "Have safe travels captain."`
