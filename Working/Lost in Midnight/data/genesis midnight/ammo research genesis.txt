outfitter "Midnight Genesis Flak Shell Restock"
	to sell
		has "gamerule: universal ammo restocking"
		or
			has "license: Genesis Flak Shell Access Temporary"
			has "license: Genesis Flak Shell Tech"
		or
			has "outfit: Genesis Flak Turret"
			has "outfit: Genesis Flak Shell Crate"
			has "outfit: Genesis Flak Shell Depot"
	location
		attributes "outfitter"
	stock
		"Genesis Flak Shell"

mission "Midnight Tech: Genesis Flak Shell Access Permanent"
	entering
	invisible
	destination Earth
	to offer
		has "license: Genesis Flak Shell Access Temporary"
		not "license: Genesis Flak Shell Tech"
	on offer
		clear "license: Genesis Flak Shell Access Temporary"
		set "license: Genesis Flak Shell Tech"
		fail

mission "Midnight Research: Genesis Flak Shell Tech"
	entering
	invisible
	destination Earth
	to offer
		has "outfit (flagship installed): Genesis Flak Shell"
		has "outfit (flagship installed): Ammo Basics"
		has "outfit (flagship installed): Research Data"
		has "outfit (flagship installed): Research Data (Chem)"
		has "outfit (flagship installed): Research Data (Phys)"
		has "license: Heavy Rocket Tech"
		not "license: Genesis Flak Shell Access Temporary"
		not "license: Genesis Flak Shell Tech"
		random < 5 + ( "flagship attribute: research bonus" / 10 )
	on offer
		outfit "Research Data" -1
		outfit "Research Data (Chem)" -1
		outfit "Research Data (Phys)" -1
		set "license: Genesis Flak Shell Tech"
		conversation
			scene "outfit/tech genesis flak shell"
			`You have successfully completed Genesis Flak Shell Tech research. This makes it possible for any outfitter to offer Genesis Flak Shells even if it is normally not stocked there.`
			``
			`	(Lost 1 Research Data)`
			`	(Lost 1 Research Data (Chem))`
			`	(Lost 1 Research Data (Phys))`
			`	(Gained Genesis Flak Shell Tech)`
				decline
