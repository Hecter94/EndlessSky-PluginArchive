mission "Successors: Spectators"
	job
	repeat
	name "Successor Spectators to <planet>"
	description "These <bunks> spectators wish to attend a performance on <date> at <destination>. They will pay you <payment> to take them there in time to watch the show."
	deadline
	passengers 2 5 .5
	to offer
		has "known to the successors"
		random < 10
	to fail
		has "Successors: Unified Defense: offered"
	source
		attributes "successor"
		not attributes "crime" "quiet" "uninhabited"
	destination
		attributes "successor"
		not attributes "crime" "quiet" "uninhabited" "station"
		distance 5 20
	on visit
		dialog phrase "generic passenger on visit"
	on complete
		payment 8000 160
		dialog phrase "successor workers out"
		"successor jobs" ++
