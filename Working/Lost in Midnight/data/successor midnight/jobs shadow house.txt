mission "Midnight Successors: Shadow Blackmail 1"
	job
	repeat
	name "Shadow Blackmail (minor)"
	description "Blackmail data stored on <tons> of databanks needs to get from <origin> to <planet> by <date>. You will be paid <payment> on delivery. If the authorities catch you with this, you will be fined."
	deadline
	cargo "blackmail data" 1 1
	illegal 50000 "The High Houses have confiscated your slanderous data of a highly respected official. A small fine has been levied against you to discourage this behavior."
	to offer
		has "license: Shadow House"
		random < 30
	to fail
		has "Successors: Unified Defense: offered"
	source Vade-Kella-Kaska
	destination
		attributes "successor"
		attributes "wealthy" "urban" "station"
		not attributes "uninhabited" "crime"
		distance 3 20
	on visit
		dialog phrase "generic cargo on visit"
	on abort
		"reputation: Shadow House" --
		dialog "You failed to deliver the blackmail data. Shadow House will not be pleased to learn of this."
	on fail
		"reputation: Shadow House" --
		dialog "You failed to deliver the blackmail data. Shadow House will not be pleased to learn of this."
	on complete
		payment
		payment 10000 1000
		dialog phrase "shadow house delivery"
		"reputation: Shadow House" ++

mission "Midnight Successors: Void Jelly Intro"
	minor
	invisible
	to offer
		or
			has "outfit (flagship installed): Voidjelly"
			has "outfit (flagship installed): Voidfish"
			has "outfit (flagship installed): Voidlunker"
			has "outfit (flagship installed): Integral Wood"
			has "outfit (flagship installed): Jamfruit"
			has "outfit (flagship installed): Pollen"
	on offer
		conversation
			scene "outfit/voidjelly"
			`A Shadow House agent slithers out of the shadows. "Our scans shows signs that thou hast visited places where the Voidjellies dwell. Occasionally we hast use for those foul creatures. We shall add it to our requests from time to time."`
			choice
				`	"I'll look out for them."`
					goto understand
				`	"What do you use them for?"`
			`	The agent answers in a subdued tone "Throughout our exploration of the stars, we found no better source for poison that suits our purposes."`
			choice
				`	"Because of how quickly it kills?"`
			`	The Shadow House agent shudders, flashing shades of yellow while making a noise that seems like laughter. "No. Perpetual death from this poison comes slowly and painfully. 'twere a perfect tool for sending a message."`
			choice
				`	"I see."`
			label understand
			`	The agent slithers away...`
				decline

mission "Midnight Shadow House Espionage (Old Houses)"
	job
	repeat
	name "Shadow Espionage"
	description "Search for the <npc> wandering near the Old Houses. Scan its outfits and deliver the data to <destination>. Payment is <payment>. If you are caught by the Old Houses, you will be sentenced to death."
	source Vade-Kella-Kaska
	destination
		government "New Houses"
		not attributes "uninhabited"
	to offer
		"reputation: Shadow House" >= 1
		has "license: Shadow House"
		has "event: midnight add tundra espionage scanner"
		random < 25
	npc "scan outfits"
		government "Old Houses"
		system
			near Uuoru-Kella 0 1
		personality target waiting staying timid secretive uninterested
			confusion 10
		fleet
			names "high houses"
			cargo 0
			variant 3
				"Kaskhorade"
			variant 2
				"Aaulqra"
			variant 1
				"Seiitej"
		dialog "You've scanned the <npc>. Return to <destination> for your reward."
		on "scan outfits"
			outfit "Old House Secrets" 1
	on visit
		dialog "You have not brought scans of Old House Secrets. Return when you have the data."
	on fail
		outfit "Old House Secrets" -1
		"reputation: Shadow House" --
		dialog "You failed to deliver the scanner data. Shadow House will not be pleased to learn of this."
	on abort
		outfit "Old House Secrets" -1
		"reputation: Shadow House" --
		dialog "You failed to deliver the scanner data. Shadow House will not be pleased to learn of this."
	on complete
		outfit "Old House Secrets" -1
		"reputation: Shadow House" ++
		payment 300000
		dialog `Your contact is pleased with the scanner data. You receive <payment> for your efforts.`
