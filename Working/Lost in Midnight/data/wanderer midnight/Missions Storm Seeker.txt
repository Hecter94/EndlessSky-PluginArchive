mission "Storm Seeker Raider 2"
	name "Salvage: Raider"
	description "Bring 145 crew and a Jump Drive to <destination>. (The outfit will automatically be removed if you bring one.)"
	source
		not planet "Season's End"
		not attributes "uninhabited"
		attributes "dirt belt" "south"
	destination "Season's End"
	passengers 145
	to offer
		has "Storm Seeker Raider 1: failed"
		has "outfit (cargo): Jump Drive"
		or
			not "Randomize Storm Seeker Raider 2"
			and
				has "Randomize Storm Seeker Raider 2"
				random < 20
	on offer
		conversation
			scene "thumbnail/raider"
			`Do you want to use one of your jump drives to salvage the abandoned pirate raider on Season's End?`
			choice
				`	Yes. I'll also need to transport 145 crew to the station to operate the salvaged ship.`
					accept
				`	Maybe another time.`
					defer
				`	Meh. I don't really want to go through the trouble.`
					decline
	on defer
		set "Randomize Storm Seeker Raider 2"
	on visit
		conversation
			scene "thumbnail/raider"
			`You don't have enough crew members or spare Jump Drives to salvage this ship.`
	to complete
		has "outfit (cargo): Jump Drive"
	on complete
		give ship "Korath Raider (Torpedo)" "Bloody End"
		outfit "Jump Drive" -1
		conversation
			scene "thumbnail/raider"
			`As soon as you arrive, your crew gets to work on installing a jump drive onto the abandoned raider and cleaning out the previous... mess. Once installed, the ship is ready to launch.`
			


mission "Storm Seeker Shallow Pond"
	minor
	name "Salvage: Shallow Pond"
	description "Bring 1 new crewmember and a spare Jump Drive to <destination>. (The outfit will automatically be removed if you bring one.)"
	repeat
	source "Season's End"
	to offer
		or
			has "Storm Seeker Raider 2: done"
			has "Storm Seeker Raider 2: declined"
		or
			and
				not "randomize Storm Seeker Shallow Pond"
				random < 70
			and
				has "randomize Storm Seeker Shallow Pond"
				random < 30 - ( 3 * "Storm Seeker Shallow Pond: done" )
	on offer
		conversation
			scene "scene/scene-shallow pond"
			`After several hours of searching, you find a ship that's almost salvageable. All it requires is a Jump Drive and a crew.`
			choice
				`Salvage the ship.`
				`Do not salvage the ship.`
					decline
			`	You'll have to recruit one crew member to pilot this ship. You'll also need an extra Jump Drive.`
					accept

	on accept
		clear "Storm Seeker Shallow Pond Crew: offered"
	on visit
		conversation
			scene "scene/scene-shallow pond"
			`You don't have enough crew members or spare Jump Drives to salvage this ship.`
	on decline
		set "randomize Storm Seeker Shallow Pond"
	to complete
		has "outfit (cargo): Jump Drive"
		has "Storm Seeker Shallow Pond Crew: done"
	on complete
		outfit "Jump Drive" -1
		give ship "Shallow Pond (Jump)"
		give ship "Flycatcher"
		set "randomize Storm Seeker Shallow Pond"

			scene "scene/scene-shallow pond"
			`As soon as you arrive, your crew gets to work on installing a jump drive onto the salvaged ship. Once installed, the ship is ready to launch.`
