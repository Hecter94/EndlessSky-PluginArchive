mission "Omnis: All Plugin Variants"
	invisible
	repeat
	to offer
		has "omnis: plugin integration"
		has "global: Omnis: Shipyard VariantsAll"
	on offer
		event "omnis: all plugin variants integration"

mission "Omnis: No Plugin Variants"
	invisible
	repeat
	to offer
		or
			not "omnis: plugin integration"
			and
				not "global: Omnis: Shipyard VariantsAll"
				not "global: Omnis: Shipyard Variants"
	on offer
		event "omnis: remove plugin variants integration"

mission "Omnis: Limited Plugin Variants"

		not "global: Omnis: Shipyard VariantsAll"
		has "global: Omnis: Shipyard Variants"

		event "omnis: limited plugin variants integration"
