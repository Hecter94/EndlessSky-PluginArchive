mission "Swizzle 0"
	repeat
	source Painter
	on offer
		conversation
			`	Swizzle 0 detected. Are you sure you want to apply the new Colour to your Fleet?`
			choice
				`	Yes, paint it.`
					accept
				`	Don't paint now.`
					defer

	on accept
		outfit "Swizzle 0(none)" -1
		event "Swizzeled0"
		fail "Swizzle 0"
