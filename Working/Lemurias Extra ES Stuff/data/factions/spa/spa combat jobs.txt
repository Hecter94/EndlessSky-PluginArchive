mission "Destroy CSSF Patrol, Small [1]"
	name "Destroy small CSSF patrol"
	description "A CSSF patrol has been disrupting plushie shipments near <system>. Destroy them for <payment>."
	job
	repeat
	to offer
		"random" > 80
		"combat rating" > 1500
	source
		government "Sayari Plushie"
	on complete
		dialog phrase "generic fleet bounty hunting payment dialog"
		payment 800000

	npc kill
		government "Yuyuko Saigyouji (Hostile)"
		system
			distance 1 3
		fleet cssf.small
		personality heroic staying nemesis target
		dialog phrase "generic hunted bounty fleet eliminated dialog"
