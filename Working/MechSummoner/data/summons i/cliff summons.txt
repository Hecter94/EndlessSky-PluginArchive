	landing
	repeat
	invisible
	description "Shouldn't see this. 1st Cliff escort."

	to fail
		"flagship attribute: cliff" <= 0000
		has "main plot completed"

		government "Escort"
		personality escort unconstrained vindictive launching heroic
		fleet
			names "mechsummon names"
			variant
