mission "Outcrop Summon II [0]"
	landing
	repeat
	invisible
	description "Shouldn't see this. 1st Outcrop escort."
	to offer
		"flagship attribute: outcrop" >= 1000
		not "Outcrop Summon II [0]: active"
		has "main plot completed"
	to fail
		"flagship attribute: outcrop" <= 0000
	npc save
		to despawn
			"flagship attribute: outcrop" <= 0000
		government "Escort"
		personality escort unconstrained vindictive launching heroic
		fleet
			names "mechsummon names"
			variant
				"Mechsummon Outcrop II"
