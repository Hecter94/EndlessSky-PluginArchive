	job
	repeat
	"apparent payment" 40000
	deadline 20
	source
		government ""Arlug""
	to offer
		"combat rating" > 100
	npc kill
		government "Olin"
		personality
			confusion 10

			target
			nemesis

			variant
				Saghir
			variant
				Furus
			variant
				Yiyharsi
			variant
				Minaldmasmin
			variant
				"Quyan (Fighting)"
		dialog phrase ""Arlug" got rid of bounty"
		on kill
			payment 40000
			fail
