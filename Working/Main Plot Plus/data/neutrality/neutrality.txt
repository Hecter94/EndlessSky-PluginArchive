            "The Book of Skadenga: declined" * 8 + "event: timothy radrickson forever" * 7 + "event: terraforming timer 2" * 6 + "FW Commitment: offered" * 5 + "Shady passenger transport 3 - double cross: done" * 4 + "Adelita 5: done" * 3 + "There Might Be Riots part 3B: done" * 2 + "Pookie, Part 5: declined" * 2 + "Pirate Blockade: done" * 2 + "Turn in Tarquin: declined" * 2 + "Pact Recon 3: done" * 2 + "Humanitarian 2: done" + "Smuggler's Den 2: done" + "Expedition to Hope 2: done" + "First Contact: Quarg: declined" + "Lost Boy 3: done" + "Hauler VI cargo prototype 3: done"  >= 12
            "combat rating" > 8103
            "net worth" >= 500000000

            has "outfit (all): Mug"
            has "license: Remnant Capital"
            has "license: City-Ship"
            has "landing access: Hai-home"
            has "hai space start"
            "year" >= 3023
    on offer
        conversation
            `As you enter the bar, a man approaches you.`
            `"So, about that war between the Free Worlds and the Republic. Whose side are you on?"`
            label "whose side are you on"
            choice
                `"The Free Worlds!"`
                    goto "free worlds"
                `"The Republic!"`
                    goto "republic"
                `"The Syndicate!"`
                    goto "syndicate"
                `"Actually, none of them."`
                    goto "neutral"
                    to display
                        not "claimed neutrality"
                `No, I really am neutral.`
                    to display

            `"Huh. Well, I admire your resolve. Can't go wrong with that."`

                goto "not neutral"

            label "syndicate"

                goto "not neutral"

            label "neutral"
            action
                set "claimed neutrality"
            `The man looks taken aback.`
            `"No sides, eh? You have to pick a side!"`
                goto "whose side are you on"

            label "not neutral"
            action
                clear "neutral pacifist"
                clear "neutral mercenary"
                clear "neutral merchant"

            label "end"
            action
                clear "claimed neutrality"
            `The man leaves without another word.`
