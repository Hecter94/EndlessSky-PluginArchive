mission "Main Plot Plus Hazehai Patch (Hidden)"
    invisible
    landing
    to offer
        has "event: reveal Hazehai"
        has "enabled main plot soft block"
        not "main plot completed"
        has "installed plugin: Lost-in-Midnight"
    on offer
        event "main plot plus: hazehai reveal patch - hidden"
