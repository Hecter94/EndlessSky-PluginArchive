# Fuel Exportation Jobs
mission empire-fuel-transport[0]
	job
	repeat
	name "Fuel Export to <planet>"
	description "Transport 2 Fuel Capacitor full of non-refined oil to <destination> before the date of <date>. Payment is <payment>."
	deadline 4 2
	destination
		not government "Empire Rebels" "Kor Mereti" "Kor Efret" "Kor Sestor" "Hai (Unfettered)" "Pirate"
		distance 2 4
	source
		government "Cromha Empire" "Mineria colonnies"
	to offer
		random < 5
		"reputation: Cromha Empire" > 100
	on accept
		outfit "Fuel Capacitor" +2
		dialog "An mineria colonnies guy gives you the 2 Fuel Capacitor and leaves your ship."
	on complete
		outfit "Fuel Capacitor" -2
		"reputation: Cromha Empire" += 10
		"reputation: Mineria colonnies" += 20
		dialog "You give the Fuel Capacitors to a mineria colonny guy and get payed <payment>."
		payment
		payment 50000 3
